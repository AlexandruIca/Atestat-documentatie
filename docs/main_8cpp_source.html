924 void verifica_evenimente()
925 {
926  SDL_Event eveniment;
927 
928  auto una_din_sageti = [](SDL_Event const& eveniment)
929  {
930  return eveniment.key.keysym.scancode == SDL_SCANCODE_UP
931  || eveniment.key.keysym.scancode == SDL_SCANCODE_DOWN
932  || eveniment.key.keysym.scancode == SDL_SCANCODE_LEFT
933  || eveniment.key.keysym.scancode == SDL_SCANCODE_RIGHT;
934  };
935 
936  auto una_din_hjkl = [](SDL_Event const& eveniment)
937  {
938  return eveniment.key.keysym.scancode == SDL_SCANCODE_J
939  || eveniment.key.keysym.scancode == SDL_SCANCODE_H
940  || eveniment.key.keysym.scancode == SDL_SCANCODE_K
941  || eveniment.key.keysym.scancode == SDL_SCANCODE_L;
942  };
943 
944  while(SDL_PollEvent(&eveniment)) {
945  switch(eveniment.type) {
946  case SDL_QUIT:
947  global::fereastra_inchisa = true;
948  break;
949  case SDL_KEYDOWN:
950  global::taste_apasate[eveniment.key.keysym.scancode] = true;
951 
952  // Daca e tasta apasata repetat
953  if((una_din_sageti(eveniment) || una_din_hjkl(eveniment))
954  && eveniment.key.repeat != 0) {
955  StadiulJocului::timp_asteptare_maxim = 50LL;
956  }
957 
958  break;
959  case SDL_KEYUP:
960  StadiulJocului::timp_asteptare_maxim = 150LL;
961  break;
962  default:
963  break;
964  }
965  }

968 bool Joc::muta_omida(StadiulJocului& stadiu)
969 {
970  Punct pozitie_cap = stadiu.pozitii_omida.front();
971  pozitie_cap.i += directie[stadiu.directia_omizii].i;
972  pozitie_cap.j += directie[stadiu.directia_omizii].j;
973 
974  if(pozitie_cap.i < 0 || pozitie_cap.i >= StadiulJocului::inaltime_teren) {
975  return false;
976  }
977  if(pozitie_cap.j < 0 || pozitie_cap.j >= StadiulJocului::lungime_teren) {
978  return false;
979  }
980 
981  PosibilitateTeren& ce_urmeaza = stadiu.teren[pozitie_cap.i][pozitie_cap.j];
982 
983  switch(ce_urmeaza) {
984  case PosibilitateTeren::OMIDA: {
985  return false;
986  }
987  case PosibilitateTeren::FRUNZA: {
988  Mix_PlayChannel(-1, m_sunet_frunza, 0);
989  ++stadiu.scor;
990 
991  stadiu.pozitie_frunza = genereaza_pozitie_noua(stadiu);
992  marcheaza_frunza(stadiu);
993 
994  ce_urmeaza = PosibilitateTeren::OMIDA;
995  stadiu.pozitii_omida.push_front(pozitie_cap);
996  return true;
997  }
998  case PosibilitateTeren::POWERUP: {
999  Mix_PlayChannel(-1, m_sunet_powerup, 0);
1000  stadiu.scor += 10;
1001 
1002  ce_urmeaza = PosibilitateTeren::OMIDA;
1003  stadiu.pozitii_omida.push_front(pozitie_cap);
1004  stadiu.exista_powerup = false;
1005 
1006  break;
1007  }
1008  case PosibilitateTeren::GOL: {
1009  ce_urmeaza = PosibilitateTeren::OMIDA;
1010  stadiu.pozitii_omida.push_front(pozitie_cap);
1011 
1012  const Punct coada = stadiu.pozitii_omida.back();
1013  stadiu.teren[coada.i][coada.j] = PosibilitateTeren::GOL;
1014  stadiu.pozitii_omida.pop_back();
1015 
1016  return true;
1017  }
1018  }
1019 
1020  return true;
