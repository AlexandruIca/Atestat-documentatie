896  std::uniform_int_distribution<int> interval_inaltime{ 0, stadiu.inaltime_teren - 1 };
897  std::uniform_int_distribution<int> interval_lungime{ 0, stadiu.lungime_teren - 1 };
898 
899  Punct pozitie = {

901  interval_lungime(generator_random)

903 
904  while(stadiu.teren[pozitie.i][pozitie.j] != PosibilitateTeren::GOL) {
905  pozitie.i = interval_inaltime(generator_random);
906  pozitie.j = interval_lungime(generator_random);
907  }

948 void verifica_evenimente()
949 {
950  SDL_Event eveniment;
951 
952  auto una_din_sageti = [](SDL_Event const& eveniment)
953  {
954  return eveniment.key.keysym.scancode == SDL_SCANCODE_UP
955  || eveniment.key.keysym.scancode == SDL_SCANCODE_DOWN
956  || eveniment.key.keysym.scancode == SDL_SCANCODE_LEFT
957  || eveniment.key.keysym.scancode == SDL_SCANCODE_RIGHT;
958  };
959 
960  auto una_din_hjkl = [](SDL_Event const& eveniment)
961  {
962  return eveniment.key.keysym.scancode == SDL_SCANCODE_J
963  || eveniment.key.keysym.scancode == SDL_SCANCODE_H
964  || eveniment.key.keysym.scancode == SDL_SCANCODE_K
965  || eveniment.key.keysym.scancode == SDL_SCANCODE_L;
966  };
967 
968  while(SDL_PollEvent(&eveniment)) {
969  switch(eveniment.type) {
970  case SDL_QUIT:
971  global::fereastra_inchisa = true;
972  break;
973  case SDL_KEYDOWN:
974  global::taste_apasate[eveniment.key.keysym.scancode] = true;
975 
976  // Daca e tasta apasata repetat
977  if((una_din_sageti(eveniment) || una_din_hjkl(eveniment))
978  && eveniment.key.repeat != 0) {
979  StadiulJocului::timp_asteptare_maxim = 50LL;
980  }
981 
982  break;
983  case SDL_KEYUP:
984  StadiulJocului::timp_asteptare_maxim = 150LL;
985  break;
986  default:
987  break;
988  }
989  }
